```python
group_by_year = df.loc[:, ['year', 'revenue', 'profit']].groupby('year')
avgs = group_by_year.mean()
x = avgs.index
y1 = avgs.profit
def plot(x, y, ax, title, y_label):
    ax:set_title(title)
    ax:set_ylabel(y_label)
    ax.plot(x, y)
    ax.margins(x = 0, y = 0)
```
